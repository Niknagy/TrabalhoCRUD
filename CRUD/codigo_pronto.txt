### db_setup.py

```python
import sqlite3

def create_database():
    conn = sqlite3.connect('tasks.db')
    cursor = conn.cursor()
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS tasks (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            employee TEXT NOT NULL,
            function TEXT NOT NULL,
            location TEXT NOT NULL,
            task TEXT NOT NULL,
            priority TEXT NOT NULL,
            status TEXT NOT NULL,
            start_date TEXT,
            end_date TEXT,
            responsible TEXT NOT NULL,
            created_date TEXT DEFAULT CURRENT_TIMESTAMP
        )
    ''')
    conn.commit()
    conn.close()

if __name__ == "__main__":
    create_database()
```

### crud.py
```python
import sqlite3

def insert_task(employee, function, location, task, priority, status, start_date, end_date, responsible):
    conn = sqlite3.connect('tasks.db')
    cursor = conn.cursor()
    cursor.execute('''
        INSERT INTO tasks (employee, function, location, task, priority, status, start_date, end_date, responsible)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    ''', (employee, function, location, task, priority, status, start_date, end_date, responsible))
    conn.commit()
    conn.close()

def get_all_tasks():
    conn = sqlite3.connect('tasks.db')
    cursor = conn.cursor()
    cursor.execute('SELECT * FROM tasks')
    tasks = cursor.fetchall()
    conn.close()
    return tasks

def update_task(id, employee, function, location, task, priority, status, start_date, end_date, responsible):
    conn = sqlite3.connect('tasks.db')
    cursor = conn.cursor()
    cursor.execute('''
        UPDATE tasks SET employee=?, function=?, location=?, task=?, priority=?, status=?, start_date=?, end_date=?, responsible=?
        WHERE id=?
    ''', (employee, function, location, task, priority, status, start_date, end_date, responsible, id))
    conn.commit()
    conn.close()

def delete_task(id):
    conn = sqlite3.connect('tasks.db')
    cursor = conn.cursor()
    cursor.execute('DELETE FROM tasks WHERE id=?', (id,))
    conn.commit()
    conn.close()

def search_tasks(query):
    conn = sqlite3.connect('tasks.db')
    cursor = conn.cursor()
    cursor.execute('SELECT * FROM tasks WHERE task LIKE ? OR employee LIKE ?', ('%' + query + '%', '%' + query + '%'))
    tasks = cursor.fetchall()
    conn.close()
    return tasks
```

### main.py
```python
import db_setup
import crud

db_setup.create_database()

def menu():
    while True:
        print("\nSistema CRUD de Tarefas")
        print("1. Inserir tarefa")
        print("2. Listar tarefas")
        print("3. Atualizar tarefa")
        print("4. Excluir tarefa")
        print("5. Buscar tarefas")
        print("6. Sair")
        choice = input("Escolha uma opção: ")
        if choice == '1':
            employee = input("Funcionário: ")
            function = input("Função: ")
            location = input("Local: ")
            task = input("Tarefa: ")
            priority = input("Prioridade: ")
            status = input("Status: ")
            start_date = input("Data de início (YYYY-MM-DD): ")
            end_date = input("Data de término (YYYY-MM-DD): ")
            responsible = input("Responsável: ")
            crud.insert_task(employee, function, location, task, priority, status, start_date, end_date, responsible)
            print("Tarefa inserida!")
        elif choice == '2':
            tasks = crud.get_all_tasks()
            if tasks:
                for t in tasks:
                    print(f"ID: {t[0]}, Funcionário: {t[1]}, Função: {t[2]}, Local: {t[3]}, Tarefa: {t[4]}, Prioridade: {t[5]}, Status: {t[6]}, Início: {t[7]}, Fim: {t[8]}, Responsável: {t[9]}, Criado: {t[10]}")
            else:
                print("Nenhuma tarefa encontrada.")
        elif choice == '3':
            id = int(input("ID da tarefa a atualizar: "))
            employee = input("Novo Funcionário: ")
            function = input("Nova Função: ")
            location = input("Novo Local: ")
            task = input("Nova Tarefa: ")
            priority = input("Nova Prioridade: ")
            status = input("Novo Status: ")
            start_date = input("Nova Data de início (YYYY-MM-DD): ")
            end_date = input("Nova Data de término (YYYY-MM-DD): ")
            responsible = input("Novo Responsável: ")
            crud.update_task(id, employee, function, location, task, priority, status, start_date, end_date, responsible)
            print("Tarefa atualizada!")
        elif choice == '4':
            id = int(input("ID da tarefa a excluir: "))
            crud.delete_task(id)
            print("Tarefa excluída!")
        elif choice == '5':
            query = input("Buscar por tarefa ou funcionário: ")
            tasks = crud.search_tasks(query)
            if tasks:
                for t in tasks:
                    print(f"ID: {t[0]}, Funcionário: {t[1]}, Função: {t[2]}, Local: {t[3]}, Tarefa: {t[4]}, Prioridade: {t[5]}, Status: {t[6]}, Início: {t[7]}, Fim: {t[8]}, Responsável: {t[9]}, Criado: {t[10]}")
            else:
                print("Nenhuma tarefa encontrada.")
        elif choice == '6':
            break
        else:
            print("Opção inválida")

if __name__ == "__main__":
    menu()
```

### app.py
```python
from flask import Flask, render_template, request, redirect, url_for
import crud
import db_setup

app = Flask(__name__)
db_setup.create_database()

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/insert', methods=['GET', 'POST'])
def insert():
    if request.method == 'POST':
        crud.insert_task(
            request.form['employee'],
            request.form['function'],
            request.form['location'],
            request.form['task'],
            request.form['priority'],
            request.form['status'],
            request.form['start_date'],
            request.form['end_date'],
            request.form['responsible']
        )
        return redirect(url_for('list_tasks'))
    return render_template('insert.html')

@app.route('/list')
def list_tasks():
    tasks = crud.get_all_tasks()
    return render_template('list.html', tasks=tasks)

@app.route('/update/<int:id>', methods=['GET', 'POST'])
def update(id):
    if request.method == 'POST':
        crud.update_task(
            id,
            request.form['employee'],
            request.form['function'],
            request.form['location'],
            request.form['task'],
            request.form['priority'],
            request.form['status'],
            request.form['start_date'],
            request.form['end_date'],
            request.form['responsible']
        )
        return redirect(url_for('list_tasks'))
    task = crud.get_all_tasks()
    task = next((t for t in task if t[0] == id), None)
    return render_template('update.html', task=task)

@app.route('/delete/<int:id>', methods=['GET', 'POST'])
def delete(id):
    if request.method == 'POST':
        crud.delete_task(id)
        return redirect(url_for('list_tasks'))
    task = crud.get_all_tasks()
    task = next((t for t in task if t[0] == id), None)
    return render_template('delete.html', task=task)

if __name__ == '__main__':
    app.run(debug=True)
```
